11   @cython.cdivision(True)
12   def PMI(word_A, word_B, bow, bowB, px_bow=None, pxpy_bow=None, lower=True):
13       """Calculates PMI for wordA and wordB in BOW"""
14       if lower:
15           word_A, word_B = word_A.lower(), word_B.lower()
16       number_of_sentsA = len(bow)
17       number_of_sentsB = len(bowB)
18       if number_of_sentsA == 0 or number_of_sentsB == 0:
19           return
20   
21       if px_bow is not None:
22           p_A, p_B = px_bow[word_A], px_bow[word_B]
23       else:
24           rel_word_countA = bow.sum(axis=0) / number_of_sentsA
25           rel_word_countB = bowB.sum(axis=0) / number_of_sentsB
26           p_A, p_B = rel_word_countA[word_A], rel_word_countB[word_B]
27   
28       coincidence = list()
29       for i in range(min(len(bow.index),len(bowB.index))):
30           coincidence.append(bow.iloc[i].loc[word_A] * bowB.iloc[i].loc[word_B] > 0)
31       p_A_B = sum(coincidence) / (min(number_of_sentsA,number_of_sentsB))
32   
33       if pxpy_bow is not None:
34           pApB = pxpy_bow.loc[word_A,word_B]
35       else:
36           pApB = p_A*p_B
37       if pApB == 0:
38           return 0 # should be numpy.nan
39       if p_A_B == 0:
40           return 0 #0 works better than -numpy.inf (WHY!?)
41       return numpy.log(p_A_B/pApB)

43   def get_collocates(query_word, bow, bowB, top_num=None, lower=True, px_bow=None, pxpy_bow=None):
44       """Return dict of query_word's collocates."""
45       collocates = dict()
46       for word in bow:
47           score = PMI(query_word, word, bow, bowB, lower=lower,px_bow=px_bow, pxpy_bow=pxpy_bow)
48           if score > 0:
49               collocates[word] = score
50       collocates = dict(sorted(collocates.items(), key=lambda item: item[1], reverse=True))
51       if top_num:
52           top_n = dict()
53           i = 0
54           for k, v in collocates.items():
55               if i == top_num or i == len(collocates):
56                   return top_n
57               top_n[k] = v
58               i += 1
59       return collocates

62   @cython.cdivision(True)
63   def calculate_word_similarity(wordA, wordB, bow, given_collocates=None,
64                                 lower=True, correction_element=lambda x,y: -((x-y)**2)/50, px_bow=None, pxpy_bow=None):
65       """Calculates word similarity"""
66   
67       wordA_collocates = get_collocates(wordA, bow, lower=lower, px_bow=px_bow, pxpy_bow=pxpy_bow)
68       if given_collocates:
69           wordB_collocates = given_collocates
70       else:
71           wordB_collocates = get_collocates(wordB, bow, lower=lower, px_bow=px_bow, pxpy_bow=pxpy_bow)
72       numerator = 0
73       denominator = sum(wordA_collocates.values()) + sum(wordB_collocates.values())
74       for collocate, collocate_score in wordA_collocates.items():
75           if collocate in wordB_collocates:
76               a, b = collocate_score, wordB_collocates[collocate]
77               numerator += a+b+correction_element(a,b)
78       return numerator/denominator

81   def get_topN_in_dict(dictionary, topn, i=1, j=None, reverse=True):
82       if j:
83           sorted_dict = dict(sorted(dictionary.items(), key=lambda item: item[i][j], reverse=reverse))
84       else:
85           sorted_dict = dict(sorted(dictionary.items(), key=lambda item: item[i], reverse=reverse))
86       top_n = dict()
87       i = 0
88       for k, v in sorted_dict.items():
89           if i == topn or i == len(sorted_dict):
90               return top_n
91           top_n[k] = v
92           i += 1
93       return top_n

96   def strip_tag_name(element_tag):
97       t = element_tag
98       i = t.find("}")
99       if i != -1:
100         t = t[i+1:]
101      return t

108  def get_parallel_sents(word, lang_element, sent_element, orig_lan, targ_lan, input_file, output_file, set_limit=None,
109                              random_sample=False):
110      context = ET.iterparse(input_file, events=('start', 'end'))
111      new_dict = dict()
112      found_word = None
113      if random_sample and set_limit:
114          old_limit = set_limit
115          set_limit *= 4
116      if output_file:
117          first_line = True
118          with open(output_file, 'w', encoding='utf-8') as outfile:
119              outfile.write(f'Phrases in {orig_lan} containing "{word}" & and their aligned phrases in {targ_lan}\n\n')
120              for index, (event, elm) in enumerate(context):
121                  if index == 0:
122                      root = elm
123                  element_tag = strip_tag_name(elm.tag)
124                  if event == 'start':
125                      if element_tag == lang_element:
126                          if elm.attrib[list(elm.attrib.keys())[0]] == orig_lan:
127                              original_lan = True
128                              target_lan = False
129                          if elm.attrib[list(elm.attrib.keys())[0]] == targ_lan:
130                              target_lan = True
131                              original_lan = False
132  
133                      if element_tag == sent_element:
134                          if elm.text:
135                              text_without_punct = remove_punct(elm.text)
136                              if original_lan:
137                                  if word in text_without_punct.split():
138                                      new_dict[elm.text] = None
139                                      found_word = elm.text
140                              elif target_lan and found_word:
141                                  new_dict[found_word] = elm.text
142                                  found_word = None
143                                  if set_limit:
144                                      if len(new_dict) >= set_limit:
145                                          if random_sample:
146                                              new_dict = dict(random.sample(new_dict.items(), old_limit))
147                                          break
148                              content_to_write = elm.text if first_line else '\n' + elm.text
149                              outfile.write(content_to_write)
150                              first_line = False
151  
152                  if event == 'end':
153                      if element_tag == 'tu':
154                          root.clear()
155      else:
156          for index, (event, elm) in enumerate(context):
157              if index == 0:
158                  root = elm
159              element_tag = strip_tag_name(elm.tag)
160              if event == 'start':
161                  if element_tag == lang_element:
162                      if elm.attrib[list(elm.attrib.keys())[0]] == orig_lan:
163                          original_lan = True
164                          target_lan = False
165                      if elm.attrib[list(elm.attrib.keys())[0]] == targ_lan:
166                          target_lan = True
167                          original_lan = False
168  
169                  if element_tag == sent_element:
170                      if elm.text:
171                          text_without_punct = remove_punct(elm.text)
172                          if original_lan:
173                              if word in text_without_punct.split():
174                                  new_dict[elm.text] = None
175                                  found_word = elm.text
176                          elif target_lan and found_word:
177                              new_dict[found_word] = elm.text
178                              found_word = None
179                              if set_limit:
180                                  if len(new_dict) >= set_limit:
181                                      if random_sample:
182                                          new_dict = dict(random.sample(new_dict.items(), old_limit))
183                                      break
184  
185  
186              if event == 'end':
187                  if element_tag == 'tu':
188                      root.clear()
189  
190      return new_dict

193  def get_frequencies(sent_sent_dict):
194      original_freq = dict()
195      target_freq = dict()
196      for sentA, sentB in sent_sent_dict.items():
197          if sentA:
198              sentA_words = remove_punct(sentA).lower().split()
199              for wordA in sentA_words:
200                  if wordA in original_freq:
201                      original_freq[wordA] += 1
202                  else:
203                      original_freq[wordA] = 1
204          if sentB:
205              sentB_words = remove_punct(sentB).lower().split()
206              for wordB in sentB_words:
207                  if wordB in target_freq:
208                      target_freq[wordB] += 1
209                  else:
210                      target_freq[wordB] = 1
211      return original_freq, target_freq

214  def get_sent_presence(words, sents, binary=True):
215      bow = pandas.DataFrame(columns=words, index=sents, data=[[0]*len(words)]*len(sents))
216      for sent in bow.index:
217          if sent:
218              tokens = remove_punct(sent).lower().split()
219              for token in tokens:
220                  if binary:
221                      bow.loc[sent,token] = 1
222                  else:
223                      bow.loc[sent,token] += 1
224      return bow

227  def get_everpresent_words(binary_bow, n, relative=False):
228      if relative:
229          return (binary_bow.sum(axis=0)/len(binary_bow.index)).nlargest(n)
230      return binary_bow.sum(axis=0).nlargest(n)

233  def tf_idf(bow, word, text):
234      """Returns TF-IDF score of word in BOW"""
235      freq = bow[word][text]
236      no_of_documents = len(bow.index)
237      no_of_documents_with_word = sum(bow[word].apply(lambda x: x > 0))
238      tfidfscore = freq*math.log10(no_of_documents/no_of_documents_with_word)
239      return tfidfscore

242  def get_tfidfbow(bow, threshold=0, verbose=False):
243      """Transforms BoW to TF-IDF BoW. Returns BoW with TF-IDF scores instead of frequencies."""
244      if verbose:
245          print('\nCreating TF-IDF Bag of Words out of sparse Bag of Words\n')
246      bow_tfidfsc = pandas.DataFrame(columns=bow.columns, index=bow.index)
247      texts = bow.index
248      for text in texts:
249          if verbose:
250              print(f'Processing: {text}')
251          for word in bow:
252              bow_tfidfsc.loc[text, word] = tf_idf(bow, word, text)
253      return bow_tfidfsc.loc[:, (bow_tfidfsc > threshold).any(axis=0)]

256  def get_translation_candidates(word, lang_element, sent_element, orig_lan, targ_lan, Ncandidates, input_file, output_file, set_limit=None,
257                              random_sample=False):
258  
259      parallel_sents = get_parallel_sents(word, lang_element, sent_element, orig_lan, targ_lan,
260                                          input_file,
261                                          output_file, set_limit=set_limit, random_sample=random_sample)
262      # Determining frequency of each word in extracted parralel sentences
263      original, target = get_frequencies(parallel_sents)
264      target_words = [word for word in target.keys()]
265      original_words = [word for word in original.keys()]
266      # Binary BOWS
267      binary_target_bow = get_sent_presence(target_words, parallel_sents.values())
268      # BOW
269      target_bow = get_sent_presence(target_words, parallel_sents.values(), binary=False)
270      original_bow = get_sent_presence(original_words, parallel_sents.keys(), binary=False)
271      # words with highest sentence occurrence
272      everpresent_target_word = get_everpresent_words(binary_target_bow, Ncandidates, relative=True)
273      tfidf_target_bow = get_tfidfbow(target_bow)
274      topN_candidates = set(everpresent_target_word.index).union(set(get_topN_in_dict(target, Ncandidates).keys()))
275      topN_candidates_scored = dict()
276      #Determining candidates' scores
277      sumdf = target_bow.sum(axis=1)
278      probability_of_word_in_text = target_bow.loc[:,:].div(sumdf,axis=0)
279      sumdf = target_bow.sum(axis=0)
280      probability_of_word = sumdf/len(sumdf)
281      kw_score_bow = probability_of_word_in_text/probability_of_word
282  
283      for candidate in topN_candidates:
284          tf_idf_score = 0
285          kw_score = 0
286          number_of_documents = len(tfidf_target_bow.index)
287          second_metric = 0
288          for document in tfidf_target_bow.index:
289              if candidate in tfidf_target_bow.loc[document,:]:
290                  second_metric += 1
291                  tf_idf_score += tfidf_target_bow.loc[document, candidate]
292                  kw_score += kw_score_bow.loc[document, candidate]
293          PMI_score = PMI(word, candidate, original_bow, target_bow)
294          #topN_candidates_scored[candidate] = (tf_idf_score / number_of_documents, kw_score/number_of_documents, PMI_score)
295          topN_candidates_scored[candidate] = tf_idf_score / number_of_documents + kw_score/number_of_documents - PMI_score
296      return get_topN_in_dict(topN_candidates_scored,Ncandidates, reverse=False)

299  def check_translation(translation_candidates, word, lang_element, sent_element, orig_lan, targ_lan,
300                        input_file, output_file=None, set_limit=None,random_sample=False):
301      for translation_candidate in translation_candidates:
302          parallel_sents = get_parallel_sents(translation_candidate, lang_element, sent_element, targ_lan, orig_lan,
303                                              input_file,
304                                              output_file=output_file, set_limit=set_limit, random_sample=random_sample)
305  
306          # Determining frequency of each word in extracted parralel sentences
307          target, original = get_frequencies(parallel_sents)
308          original_words = [word for word in target.keys()]
309          target_words = [word for word in original.keys()]
310  
311          # Binary BOWS
312          binary_target_bow = get_sent_presence(target_words, parallel_sents.values())
313          binary_original_bow = get_sent_presence(original_words, parallel_sents.keys())
314          results = dict()
315          if word in binary_original_bow:
316              target_in_document_freq = binary_target_bow.sum(axis=0)[translation_candidate]
317              word_in_doc_freq = binary_original_bow.sum(axis=0)[word]
318              results[translation_candidate] = word_in_doc_freq/target_in_document_freq
319          else:
320              results[translation_candidate] = 0
321      return get_topN_in_dict(results, len(results))
